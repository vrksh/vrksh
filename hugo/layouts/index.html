# Fetch the page
CONTENT=$(vrk grab https://example.com/article)

# Guard: fail if content exceeds token budget
echo "$CONTENT" | vrk tok --budget 8000 \
  # Summarize with an LLM
  && echo "$CONTENT" \
     | vrk prompt --system "Summarize in 3 bullet points." \
  # Validate the response against a JSON schema
     | vrk validate --schema summary.json \
  # Store the result
     | vrk kv set "summary:latest"

Works where agents work.

vrksh ships a discovery-only MCP server. Add it to Claude Code and every tool becomes visible to your agent. Tools run as shell commands - the MCP layer handles discovery, not execution. 

{
  "mcpServers": {
    "vrksh": {
      "command": "vrk",
      "args": ["mcp"]
    }
  }
}

Or use vrk --manifest for JSON tool discovery without MCP.
